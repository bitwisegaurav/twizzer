<script>
    const upvote = document.querySelector('.upvote');
    const downvote = document.querySelector('.downvote');
    const votes = document.querySelector('.votes');
    let voteCount = Number(votes.textContent);
    let clicked = "";
    upvote.addEventListener('click', () => {
        if(clicked === ""){
            clicked = "up";
            upvote.style.color = "orange";
            voteCount++;
        }
        else if(clicked === "up"){
            clicked = "";
            upvote.style.color = "black";
            voteCount--;
        }
        else if(clicked === "down"){
            clicked = "up";
            upvote.style.color = "orange";
            downvote.style.color = "black";
            voteCount++;
        }
        votes.textContent = voteCount;
    });
    downvote.addEventListener('click', () => {
        if(clicked === ""){
            voteCount--;
            clicked = "down";
            downvote.style.color = "orange";
        }
        else if(clicked === "up"){
            voteCount--;
            clicked = "down";
            upvote.style.color = "black";
            downvote.style.color = "orange";
        }
        else if(clicked === "down"){
            clicked = "";
            downvote.style.color = "black";
            voteCount++;
        }
        votes.textContent = voteCount;
    });
</script>
